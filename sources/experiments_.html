def train_testing(_model, _split_amounts, _features, _targets):
    """By chosing one out of three classifiers as strings, the method
    calculates the average accuracy of an untrained model. The number of
    iterations used to find the mean can be set as the _split_amounts value.
    Additional, the scaling method for algorithms can be tested by choosing either
    minmax or standard scaler.

    Args:
        _model (str): string indicating what model should be tested
        _split_amounts (int): amount of interations used to determine mean accuracy
        _features (DataFrame): Features to train the classifier on
        _targets (DataFrame): Features to evaluate classifier performance

    Returns:
        _type_: _description_
    """
    _accuracy_lis = []
    if _model == 'dt':
        _clf = DecisionTreeClassifier()
    elif _model == 'kNN':
        _clf = KNeighborsClassifier()
    elif _model == 'rf':
        _clf = RandomForestClassifier()
    for i in tqdm(range(_split_amounts)):
        _rand_num = round(np.random.random()*100)
        _X_train, _X_test, _y_train, _y_test = train_test_split(_features, _targets, test_size=0.4, shuffle=True, stratify=_targets, random_state=round(_rand_num*100))
        _X_train_sc, _X_test_sc = scale_standard(_X_train, _X_test)
        _clf.fit(_X_train_sc, _y_train)
        _y_pred = _clf.predict(_X_test_sc)
        _accuracy_lis.append(accuracy_score(_y_test, _y_pred))
    _mean_accuracy = round(np.mean(_accuracy_lis),3)
    _std_accuracy = round(np.std(_accuracy_lis),3)
    print(f'Model parameters:\n{_clf.get_params()}')
    return _mean_accuracy, _std_accuracy, _accuracy_lis

def advanced_train_testing(_model, _test_range, _split_amounts, _features, _targets,_scaleMin):
    """Extended version of train_testing method. Allows for specific training on selected
    classifier variables. The selection of specific features should be provided as a list,
    which then be iterated through using _split amounts. The total effort of this method
    therefore is: feature_list_length * _split_amounts * training time

    By chosing one out of three classifiers as strings, the method
    calculates the average accuracy of an untrained model. The number of
    iterations used to find the mean can be set as the _split_amounts value

    Args:
        _model (str): string indicating what model should be tested
        _test_range (int): amount of interations used to determine mean accuracy
        _split_amounts (int): Number of iterations per feature
        _features (DataFrame): Features to train the classifier on
        _targets (DataFrame): Features to evaluate classifier performance
        _scaleMin (bool): describes if minMax or standard scaler should be used

    Returns:
        _type_: _description_
    """
    _accuracy_lis = []
    feature_train_lis = []
    for curr_test in tqdm(_test_range):
        if _model == 'dt':
            _clf = DecisionTreeClassifier(criterion = curr_test)
        elif _model == 'kNN':
            _clf = KNeighborsClassifier(n_neighbors = curr_test)
        elif _model == 'rf':
            _clf = RandomForestClassifier(n_estimators = curr_test)
        for i in tqdm(range(_split_amounts)):
            _rand_num = round(np.random.random()*1000)
            _X_train, _X_test, _y_train, _y_test = train_test_split(_features, _targets, test_size=0.4, shuffle=True, stratify=_targets, random_state=round(_rand_num*100))
            if _scaleMin == True:
                _X_train_sc, _X_test_sc = scale_minMax(_X_train, _X_test)
            else:
                _X_train_sc, _X_test_sc = scale_standard(_X_train, _X_test)
            _clf.fit(_X_train_sc, _y_train)
            _y_pred = _clf.predict(_X_test_sc)
            _accuracy_lis.append(accuracy_score(_y_test, _y_pred))
        _mean_accuracy = round(np.mean(_accuracy_lis),3)
        _std_accuracy = round(np.std(_accuracy_lis),3)
        feature_train_lis.append({curr_test : (_mean_accuracy,_std_accuracy)})
    return feature_train_lis

def evaluate_pipeline(_model_type,_resolution = 3, _features=None, _targets=None, _pipeline = None, _parameter_grid=None):
    """Implementation of nested cross validation. Takes a pipeline, description 
    of model type and data to perform nested cross validation and return the resulting
    best average performances including additional parameters.

    Args:
        _model_type (str): short description to identify model in dataframe
        _resolution (int, optional): Number of splits & iterations used nested cross validation. Defaults to 3.
        _features (DataFrame): Entire set of Training data
        _targets (DataFrame): Entire set of target values
        _pipeline (Pipeline): Pipleline including specific model to be trained
        _parameter_grid (dict): dictionary with any parameters to optimise

    Returns:
        DataFrame: Data of trained optimal model as dataframe 
    """
# Reference: https://scikit-learn.org/stable/auto_examples/model_selection/plot_nested_cross_validation_iris.html
    inner_cv = StratifiedKFold(n_splits=_resolution, shuffle=True)
    outer_cv = RepeatedStratifiedKFold(n_splits=_resolution, n_repeats=2)
    outer_results, scores, precisions, roc_base, recalls, f1_macros = [], [], [], [], [], []
    cnt = 0

    #--- outer loop ---
    for train_index, test_index in tqdm(outer_cv.split(_features,_targets)):
        X_train_outer, X_test_outer = _features.iloc[train_index], _features.iloc[test_index]
        y_train_outer, y_test_outer = _targets[train_index], _targets[test_index]

        clf = GridSearchCV(estimator=_pipeline, n_jobs=-1, param_grid=_parameter_grid, cv=inner_cv, scoring='balanced_accuracy')

        print(f'Outer Loop {cnt+1}/{_resolution+1}',end=' ')
        #--- inner loop ---
        start_time = time.time()
        cv_result = cross_validate(clf, _features, _targets, cv=outer_cv, scoring=["accuracy","balanced_accuracy", "precision_macro", "recall_macro", "f1_macro"], n_jobs=-1, return_train_score=True)

        #--- Tempoary Training ---
        clf.fit(X_train_outer, y_train_outer)
        best_model = clf.best_estimator_
        test_accuracy = balanced_accuracy_score(y_test_outer, best_model.predict(X_test_outer))
        train_accuracy = balanced_accuracy_score(y_train_outer, best_model.predict(X_train_outer))
        roc_base.append({'train_proba': best_model.predict(X_train_outer),'y_train': y_train_outer})
        outer_results.append({'train_accuracy': train_accuracy,'test_accuracy': test_accuracy})

        #--- Final Scoring ---
        scores.append(cv_result['test_balanced_accuracy'])
        precisions.append(cv_result['test_precision_macro'])
        recalls.append(cv_result['test_recall_macro'])
        f1_macros.append(cv_result['test_f1_macro'])
        end_time = time.time()
        print(f' - {round(end_time - start_time, 2)}s')
        #--- Result Data processing & wrapping ---
        cnt +=1 
    _df = edit_data(_model_type, cv_result['test_accuracy'],cv_result['test_balanced_accuracy'], cv_result['test_precision_macro'],cv_result['test_f1_macro'],cv_result['test_recall_macro'], outer_results)
    print(f"Best Score: {cv_result['test_balanced_accuracy']}\n")
    return _df, roc_base
